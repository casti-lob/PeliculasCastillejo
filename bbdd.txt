CREATE TABLE USERS
	(id 			int(8),
    password 		VARCHAR(32),
	name 		VARCHAR(200),
    nick		varchar(100),
    dateUser	DATE,
    gender		boolean,
    admin		boolean,
	CONSTRAINT pk_users PRIMARY KEY (id)
	);
    
CREATE TABLE CATEGORY
	(code_cat 		int(8), 
     name 		varchar(100),
     description    varchar(300),
     CONSTRAINT pk_category PRIMARY KEY (code_cat)
    );
CREATE TABLE ELEMENT
	(code_ele 		int(8),
    name_ele		varchar(100) unique,
    description_ele varchar(300),
    price			double(8,3),
     code_cat 		int(8),
    CONSTRAINT pk_element PRIMARY KEY (code_ele),
    CONSTRAINT fk_element FOREIGN KEY (code_cat) REFERENCES CATEGORY(code_cat) ON DELETE CASCADE
    );
    
CREATE TABLE PRODUCT
	(id_user 	int(8),
     code_ele	int(8),
     number		int(8),
     price		double(8,3),
     dateBuy 	date,
     CONSTRAINT pk_product PRIMARY KEY (id_user,code_ele,dateBuy),
     CONSTRAINT fk_product_user FOREIGN KEY (id_user) REFERENCES USERS (id) ON DELETE CASCADE,
     CONSTRAINT fk_product_element FOREIGN KEY (code_ele) REFERENCES ELEMENT (code_ele) ON DELETE CASCADE
       );
